local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local ProximityPromptService = game:GetService("ProximityPromptService")
local LocalPlayer = Players.LocalPlayer

local enemyPosition = nil
local basePosition = nil
local espObjects = { enemy = {}, base = {} }

local function createESP()
    espObjects.enemy.circle = Drawing.new("Circle")
    espObjects.enemy.circle.Thickness = 2
    espObjects.enemy.circle.Filled = true
    espObjects.enemy.circle.Color = Color3.new(1, 0, 0)
    espObjects.enemy.circle.Visible = false
    espObjects.enemy.circle.Radius = 20
    espObjects.enemy.circle.NumSides = 24
    
    espObjects.enemy.line = Drawing.new("Line")
    espObjects.enemy.line.Thickness = 2
    espObjects.enemy.line.Color = Color3.new(1, 0, 0)
    espObjects.enemy.line.Visible = false
    
    espObjects.enemy.text = Drawing.new("Text")
    espObjects.enemy.text.Size = 14
    espObjects.enemy.text.Center = true
    espObjects.enemy.text.Color = Color3.new(1, 1, 1)
    espObjects.enemy.text.Outline = true
    espObjects.enemy.text.Visible = false
    
    espObjects.base.circle = Drawing.new("Circle")
    espObjects.base.circle.Thickness = 2
    espObjects.base.circle.Filled = true
    espObjects.base.circle.Color = Color3.new(0, 1, 0)
    espObjects.base.circle.Visible = false
    espObjects.base.circle.Radius = 20
    espObjects.base.circle.NumSides = 24
    
    espObjects.base.line = Drawing.new("Line")
    espObjects.base.line.Thickness = 2
    espObjects.base.line.Color = Color3.new(0, 1, 0)
    espObjects.base.line.Visible = false
    
    espObjects.base.text = Drawing.new("Text")
    espObjects.base.text.Size = 14
    espObjects.base.text.Center = true
    espObjects.base.text.Color = Color3.new(1, 1, 1)
    espObjects.base.text.Outline = true
    espObjects.base.text.Visible = false
end

local function updateESP()
    local camera = workspace.CurrentCamera
    local playerChar = LocalPlayer.Character
    local rootPart = playerChar and playerChar:FindFirstChild("HumanoidRootPart")
    local playerPos = rootPart and rootPart.Position
    
    if enemyPosition and camera and playerPos then
        local targetScreen, onScreen = camera:WorldToViewportPoint(enemyPosition)
        local playerScreen = camera:WorldToViewportPoint(playerPos)
        
        if onScreen then
            espObjects.enemy.circle.Position = Vector2.new(targetScreen.X, targetScreen.Y)
            espObjects.enemy.circle.Visible = true
            espObjects.enemy.text.Position = Vector2.new(targetScreen.X, targetScreen.Y - 25)
            espObjects.enemy.text.Text = string.format("Enemy\n(%.0f, %.0f, %.0f)", enemyPosition.X, enemyPosition.Y, enemyPosition.Z)
            espObjects.enemy.text.Visible = true
        else
            espObjects.enemy.circle.Visible = false
            espObjects.enemy.text.Visible = false
        end
        espObjects.enemy.line.From = Vector2.new(playerScreen.X, playerScreen.Y)
        espObjects.enemy.line.To = Vector2.new(targetScreen.X, targetScreen.Y)
        espObjects.enemy.line.Visible = true
    else
        espObjects.enemy.circle.Visible = false
        espObjects.enemy.line.Visible = false
        espObjects.enemy.text.Visible = false
    end
    
    if basePosition and camera and playerPos then
        local targetScreen, onScreen = camera:WorldToViewportPoint(basePosition)
        local playerScreen = camera:WorldToViewportPoint(playerPos)
        
        if onScreen then
            espObjects.base.circle.Position = Vector2.new(targetScreen.X, targetScreen.Y)
            espObjects.base.circle.Visible = true
            espObjects.base.text.Position = Vector2.new(targetScreen.X, targetScreen.Y - 25)
            espObjects.base.text.Text = string.format("Base\n(%.0f, %.0f, %.0f)", basePosition.X, basePosition.Y, basePosition.Z)
            espObjects.base.text.Visible = true
        else
            espObjects.base.circle.Visible = false
            espObjects.base.text.Visible = false
        end
        espObjects.base.line.From = Vector2.new(playerScreen.X, playerScreen.Y)
        espObjects.base.line.To = Vector2.new(targetScreen.X, targetScreen.Y)
        espObjects.base.line.Visible = true
    else
        espObjects.base.circle.Visible = false
        espObjects.base.line.Visible = false
        espObjects.base.text.Visible = false
    end
end

local function teleportToBase()
    if not basePosition then
        return
    end
    local char = LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        char.HumanoidRootPart.CFrame = CFrame.new(basePosition)
    end
end

local function teleportToEnemy()
    if not enemyPosition then
        return
    end
    local char = LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        char.HumanoidRootPart.CFrame = CFrame.new(enemyPosition)
    end
end

local function hookProximityPrompts()
    ProximityPromptService.PromptButtonHoldEnded:Connect(function(prompt, player)
        if player ~= LocalPlayer then return end
        if prompt.Name == "Steal" or prompt.ActionText == "Steal" then
            task.wait(0.1)
            teleportToBase()
        end
    end)
end
